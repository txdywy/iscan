Actually, per D-10, Registry is `map[Layer]Probe` — one entry per layer. The qtype variation (A vs AAAA) is handled by creating separate adapter instances or by making qtype part of Opts.

Simpler approach: register one DNS adapter per resolver+domain combination. But since the Registry is keyed by Layer only, and the scanner iterates resolvers externally, register the adapter once with QType defaulting to TypeA. The scanner will instantiate adapters with different QTypes per resolver.

Actually, looking at the scanner flow: the scanner iterates resolvers, and for each resolver runs both A and AAAA queries. In the new model, the scanner would create DNS adapters with different QTypes at probe list construction time. But `init()` runs once with defaults.

Best approach: register ONE default A-record adapter. The scanner creates per-resolver adapter instances (or uses the registered one as a template). Since the Registry maps Layer->Probe, and the scanner needs multiple DNS probes (one per resolver+type), the scanner can clone the template with different QTypes.

Simplest: the init() registers the default A-record adapter. The scanner constructs per-resolver adapters by setting `Adapter{Opts: Opts{QType: mdns.TypeA/TypeAAAA, Resolver: resolver}}`. The Registry holds the template.

**File: `internal/probe/dnsprobe/adapter.go`**

```go
package dnsprobe

import (
    "context"
    "net"
    "time"

    mdns "github.com/miekg/dns"

    "iscan/internal/model"
    "iscan/internal/probe"
)

// Opts holds DNS probe configuration set at construction time.
type Opts struct {
    Resolver model.Resolver `json:"resolver,omitempty"`
    QType    uint16         `json:"qtype,omitempty"` // mdns.TypeA or mdns.TypeAAAA
    Timeout  time.Duration  `json:"timeout,omitempty"`
}

// Adapter implements probe.Probe for DNS queries.
type Adapter struct {
    Opts Opts
}

func (a *Adapter) Run(ctx context.Context, target model.Target) model.ProbeResult {
    timeout := a.Opts.Timeout
    if timeout <= 0 {
        timeout = 5 * time.Second
    }
    if a.Opts.Resolver.System {
        // System resolver path — adapted from scanner.probeDNS lines 186-213.
        start := time.Now()
        obs := model.DNSObservation{
            Resolver: a.Opts.Resolver.Name,
            Query:    target.Domain,
            Type:     mdns.TypeToString[a.Opts.QType],
        }
        network := "ip4"
        if a.Opts.QType == mdns.TypeAAAA {
            network = "ip6"
        }
        ips, err := net.DefaultResolver.LookupIP(ctx, network, target.Domain)
        obs.Latency = time.Since(start)
        if err != nil {
            obs.Error = err.Error()
            return probe.NewResult(model.LayerDNS, obs)
        }
        for _, ip := range ips {
            if a.Opts.QType == mdns.TypeA && ip.To4() != nil {
                obs.Answers = append(obs.Answers, ip.String())
            }
            if a.Opts.QType == mdns.TypeAAAA && ip.To4() == nil {
                obs.Answers = append(obs.Answers, ip.String())
            }
        }
        obs.Success = true
        obs.RCode = "NOERROR"
        return probe.NewResult(model.LayerDNS, obs)
    }
    // Custom resolver — delegate to existing dnsprobe.Probe.
    obs := Probe(ctx, a.Opts.Resolver, target.Domain, a.Opts.QType, timeout)
    return probe.NewResult(model.LayerDNS, obs)
}

func init() {
    probe.Registry[model.LayerDNS] = &Adapter{
        Opts: Opts{QType: mdns.TypeA},
    }
}
```

package httpprobe

import (
    "context"
    "time"

    "iscan/internal/model"
    "iscan/internal/probe"
)

// Opts holds HTTP probe configuration set at construction time.
type Opts struct {
    DialAddress string        `json:"dial_address,omitempty"`
    Timeout     time.Duration `json:"timeout,omitempty"`
}

// Adapter implements probe.Probe for HTTP GET requests.
type Adapter struct {
    Opts Opts
}

func (a *Adapter) Run(ctx context.Context, target model.Target) model.ProbeResult {
    timeout := a.Opts.Timeout
    if timeout <= 0 {
        timeout = 5 * time.Second
    }
    // Construct URL from target (reuse scanner.TargetURL logic):
    // scheme://domain/path (e.g., "https://example.com/")
    path := target.HTTPPath
    if path == "" {
        path = "/"
    }
    url := target.Scheme + "://" + target.Domain + path
    var obs model.HTTPObservation
    if a.Opts.DialAddress != "" {
        obs = ProbeWithAddress(ctx, url, a.Opts.DialAddress, timeout)
    } else {
        obs = Probe(ctx, url, timeout)
    }
    return probe.NewResult(model.LayerHTTP, obs)
}

func init() {
    probe.Registry[model.LayerHTTP] = &Adapter{
        Opts: Opts{Timeout: 5 * time.Second},
    }
}
